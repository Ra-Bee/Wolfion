import express from "express";
import OpenAI from "openai";
import fetch from "node-fetch";
import multer from "multer";
import pdfParse from "pdf-parse";

const app = express();
app.use(express.json({ limit: "25mb" }));

const upload = multer({ dest: "uploads/" });

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

// -------------------- AI CORE --------------------
async function askAI(messages) {
  const res = await client.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [
      {
        role: "system",
        content:
          "You are UniRab AI, a university study assistant. Be clear, structured, helpful, and student-friendly. Use live data if provided.",
      },
      ...messages,
    ],
  });

  return res.choices[0].message.content;
}

// -------------------- GOOGLE SEARCH --------------------
async function googleSearch(query) {
  try {
    const response = await fetch(
      `https://serpapi.com/search.json?q=${encodeURIComponent(query)}&api_key=${process.env.SERP_API_KEY}`
    );

    const data = await response.json();

    const results = data.organic_results
      ?.slice(0, 3)
      .map(r => `${r.title}: ${r.snippet}`)
      .join("\n");

    return results || "No results found.";
  } catch {
    return "Search failed.";
  }
}

// -------------------- SMART ROUTING --------------------
function needsLiveData(question) {
  const keywords = ["latest", "today", "news", "current", "2025", "2026"];
  return keywords.some(word => question.toLowerCase().includes(word));
}

// -------------------- MEMORY --------------------
const userMemory = {};

function getMemory(userId) {
  return userMemory[userId] || "";
}

function saveMemory(userId, memory) {
  userMemory[userId] = memory;
}

// -------------------- CHAT --------------------
app.post("/ai/chat", async (req, res) => {
  try {
    const { userId, message } = req.body;

    let context = message;

    if (needsLiveData(message)) {
      const liveData = await googleSearch(message);
      context = `Use this latest info:\n${liveData}\n\nQuestion: ${message}`;
    }

    const memory = getMemory(userId);

    const reply = await askAI([
      { role: "system", content: `User context: ${memory}` },
      { role: "user", content: context },
    ]);

    res.json({ reply });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

// -------------------- SAVE MEMORY --------------------
app.post("/ai/save-memory", (req, res) => {
  const { userId, memory } = req.body;
  saveMemory(userId, memory);
  res.json({ status: "saved" });
});

// -------------------- TEXT SUMMARY --------------------
app.post("/ai/summarize-text", async (req, res) => {
  try {
    const { text } = req.body;

    const summary = await askAI([
      {
        role: "user",
        content: `
Summarize:
- Main topic
- Key points
- Simple explanation
- Add references if possible

${text}
        `,
      },
    ]);

    res.json({ summary });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

// -------------------- URL SUMMARY --------------------
app.post("/ai/summarize-url", async (req, res) => {
  try {
    const { url } = req.body;

    const page = await fetch(url);
    const html = await page.text();

    const text = html.replace(/<[^>]*>/g, " ").slice(0, 15000);

    const summary = await askAI([
      {
        role: "user",
        content: `Summarize this article and include references:\n${text}`,
      },
    ]);

    res.json({ summary });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

// -------------------- PDF SUMMARY --------------------
app.post("/ai/summarize-pdf", upload.single("file"), async (req, res) => {
  try {
    const data = await pdfParse(req.file.buffer || req.file.path);
    const text = data.text.slice(0, 15000);

    const summary = await askAI([
      {
        role: "user",
        content: `
Summarize this PDF:
- Topic
- Key points
- Important details

${text}
        `,
      },
    ]);

    res.json({ summary });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

// -------------------- AUDIO → TEXT --------------------
app.post("/ai/transcribe", async (req, res) => {
  try {
    const { audioUrl } = req.body;

    const response = await client.audio.transcriptions.create({
      model: "gpt-4o-mini-transcribe",
      file: await fetch(audioUrl),
    });

    res.json({ text: response.text });
  } catch (e) {
    res.status(500).json({ error: e.message });
  }
});

// -------------------- SERVER --------------------
app.listen(3000, () => {
  console.log("🔥 UniRab AI FULL SYSTEM running on port 3000");
});
